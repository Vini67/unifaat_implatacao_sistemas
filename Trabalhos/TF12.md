## 🗄️ Database Schema

```sql
-- schema.sql
CREATE DATABASE IF NOT EXISTS taskmanager;
USE taskmanager;

-- Tabela de usuários
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Tabela de tarefas
CREATE TABLE tasks (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    status ENUM('pending', 'in_progress', 'completed') DEFAULT 'pending',
    priority ENUM('low', 'medium', 'high') DEFAULT 'medium',
    due_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    INDEX idx_user_status (user_id, status),
    INDEX idx_due_date (due_date)
);

-- Tabela de categorias
CREATE TABLE categories (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    color VARCHAR(7) DEFAULT '#007bff',
    user_id INT NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- Tabela de relacionamento task-category
CREATE TABLE task_categories (
    task_id INT,
    category_id INT,
    PRIMARY KEY (task_id, category_id),
    FOREIGN KEY (task_id) REFERENCES tasks(id) ON DELETE CASCADE,
    FOREIGN KEY (category_id) REFERENCES categories(id) ON DELETE CASCADE
);

-- View para estatísticas
CREATE VIEW task_stats AS
SELECT 
    u.username,
    COUNT(t.id) as total_tasks,
    SUM(CASE WHEN t.status = 'completed' THEN 1 ELSE 0 END) as completed_tasks,
    SUM(CASE WHEN t.status = 'pending' THEN 1 ELSE 0 END) as pending_tasks,
    SUM(CASE WHEN t.due_date < CURDATE() AND t.status != 'completed' THEN 1 ELSE 0 END) as overdue_tasks
FROM users u
LEFT JOIN tasks t ON u.id = t.user_id
GROUP BY u.id, u.username;
```

## 🐍 Backend API (Python Flask)

```python
# app.py
from flask import Flask, request, jsonify
from flask_cors import CORS
import pymysql
import boto3
import json
import os
from datetime import datetime

app = Flask(__name__)
CORS(app)

# Configuração do banco via Secrets Manager
def get_db_config():
    secrets_client = boto3.client('secretsmanager', region_name='us-east-1')
    secret_value = secrets_client.get_secret_value(SecretId='rds-taskmanager-credentials')
    secret = json.loads(secret_value['SecretString'])
    
    return {
        'host': secret['host'],
        'user': secret['username'],
        'password': secret['password'],
        'database': 'taskmanager',
        'charset': 'utf8mb4'
    }

def get_db_connection():
    config = get_db_config()
    return pymysql.connect(**config)

@app.route('/health', methods=['GET'])
def health_check():
    try:
        conn = get_db_connection()
        cursor = conn.cursor()
        cursor.execute("SELECT 1")
        cursor.close()
        conn.close()
        return jsonify({'status': 'healthy', 'database': 'connected'})
    except Exception as e:
        return jsonify({'status': 'unhealthy', 'error': str(e)}), 500

@app.route('/api/tasks', methods=['GET'])
def get_tasks():
    try:
        user_id = request.args.get('user_id', 1)  # Simplificado para o lab
        
        conn = get_db_connection()
        cursor = conn.cursor(pymysql.cursors.DictCursor)
        
        cursor.execute("""
            SELECT t.*, GROUP_CONCAT(c.name) as categories
            FROM tasks t
            LEFT JOIN task_categories tc ON t.id = tc.task_id
            LEFT JOIN categories c ON tc.category_id = c.id
            WHERE t.user_id = %s
            GROUP BY t.id
            ORDER BY t.created_at DESC
        """, (user_id,))
        
        tasks = cursor.fetchall()
        cursor.close()
        conn.close()
        
        return jsonify(tasks)
    except Exception as e:
        return jsonify({'error': str(e)}), 500

@app.route('/api/tasks', methods=['POST'])
def create_task():
    try:
        data = request.json
        user_id = data.get('user_id', 1)  # Simplificado
        
        conn = get_db_connection()
        cursor = conn.cursor()
        
        cursor.execute("""
            INSERT INTO tasks (user_id, title, description, status, priority, due_date)
            VALUES (%s, %s, %s, %s, %s, %s)
        """, (
            user_id,
            data['title'],
            data.get('description', ''),
            data.get('status', 'pending'),
            data.get('priority', 'medium'),
            data.get('due_date')
        ))
        
        task_id = cursor.lastrowid
        conn.commit()
        cursor.close()
        conn.close()
        
        return jsonify({'id': task_id, 'message': 'Task created successfully'})
    except Exception as e:
        return jsonify({'error': str(e)}), 500

@app.route('/api/stats', methods=['GET'])
def get_stats():
    try:
        user_id = request.args.get('user_id', 1)
        
        conn = get_db_connection()
        cursor = conn.cursor(pymysql.cursors.DictCursor)
        
        cursor.execute("""
            SELECT * FROM task_stats WHERE username = (
                SELECT username FROM users WHERE id = %s
            )
        """, (user_id,))
        
        stats = cursor.fetchone()
        cursor.close()
        conn.close()
        
        return jsonify(stats or {})
    except Exception as e:
        return jsonify({'error': str(e)}), 500

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=True)
```

## 🚀 RDS Setup Script

```bash
#!/bin/bash
# rds-setup.sh

set -e

DB_INSTANCE_ID="taskmanager-db"
DB_NAME="taskmanager"
MASTER_USERNAME="admin"
MASTER_PASSWORD="TaskManager2024!"

echo "🗄️ Configurando RDS MySQL..."

# Criar subnet group
aws rds create-db-subnet-group \
    --db-subnet-group-name taskmanager-subnet-group \
    --db-subnet-group-description "Subnet group for TaskManager RDS" \
    --subnet-ids subnet-12345678 subnet-87654321

# Criar parameter group customizado
aws rds create-db-parameter-group \
    --db-parameter-group-name taskmanager-params \
    --db-parameter-group-family mysql8.0 \
    --description "Custom parameters for TaskManager"

# Configurar parâmetros otimizados
aws rds modify-db-parameter-group \
    --db-parameter-group-name taskmanager-params \
    --parameters "ParameterName=innodb_buffer_pool_size,ParameterValue={DBInstanceClassMemory*3/4},ApplyMethod=pending-reboot"

# Criar Security Group para RDS
RDS_SG_ID=$(aws ec2 create-security-group \
    --group-name TaskManager-RDS-SG \
    --description "Security Group for TaskManager RDS" \
    --query 'GroupId' --output text)

# Permitir acesso MySQL apenas do Security Group da aplicação
aws ec2 authorize-security-group-ingress \
    --group-id $RDS_SG_ID \
    --protocol tcp \
    --port 3306 \
    --source-group sg-app12345678

# Criar instância RDS
aws rds create-db-instance \
    --db-instance-identifier $DB_INSTANCE_ID \
    --db-instance-class db.t3.micro \
    --engine mysql \
    --engine-version 8.0.35 \
    --master-username $MASTER_USERNAME \
    --master-user-password $MASTER_PASSWORD \
    --allocated-storage 20 \
    --storage-type gp2 \
    --vpc-security-group-ids $RDS_SG_ID \
    --db-subnet-group-name taskmanager-subnet-group \
    --db-parameter-group-name taskmanager-params \
    --backup-retention-period 7 \
    --preferred-backup-window "03:00-04:00" \
    --preferred-maintenance-window "sun:04:00-sun:05:00" \
    --multi-az \
    --storage-encrypted \
    --monitoring-interval 60 \
    --enable-performance-insights \
    --tags Key=Name,Value=TaskManager-Database Key=Environment,Value=Production

echo "⏳ Aguardando RDS ficar disponível..."
aws rds wait db-instance-available --db-instance-identifier $DB_INSTANCE_ID

# Obter endpoint
ENDPOINT=$(aws rds describe-db-instances \
    --db-instance-identifier $DB_INSTANCE_ID \
    --query 'DBInstances[0].Endpoint.Address' \
    --output text)

echo "✅ RDS criado com sucesso!"
echo "📍 Endpoint: $ENDPOINT"

# Criar secret no Secrets Manager
aws secretsmanager create-secret \
    --name "rds-taskmanager-credentials" \
    --description "Credentials for TaskManager RDS" \
    --secret-string "{\"username\":\"$MASTER_USERNAME\",\"password\":\"$MASTER_PASSWORD\",\"host\":\"$ENDPOINT\",\"port\":3306,\"dbname\":\"$DB_NAME\"}"

echo "🔐 Credenciais armazenadas no Secrets Manager"
```
