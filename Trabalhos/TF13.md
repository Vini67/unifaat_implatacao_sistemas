# metrics.py
import boto3
import time
import json
from datetime import datetime
from functools import wraps

class MetricsCollector:
    def __init__(self):
        self.cloudwatch = boto3.client('cloudwatch')
        self.namespace = 'UniFAAT/TaskManager'
    
    def put_metric(self, metric_name, value, unit='Count', dimensions=None):
        """Enviar métrica customizada para CloudWatch"""
        try:
            self.cloudwatch.put_metric_data(
                Namespace=self.namespace,
                MetricData=[
                    {
                        'MetricName': metric_name,
                        'Value': value,
                        'Unit': unit,
                        'Timestamp': datetime.utcnow(),
                        'Dimensions': dimensions or []
                    }
                ]
            )
        except Exception as e:
            print(f"Erro ao enviar métrica {metric_name}: {e}")
    
    def track_response_time(self, func):
        """Decorator para medir tempo de resposta"""
        @wraps(func)
        def wrapper(*args, **kwargs):
            start_time = time.time()
            try:
                result = func(*args, **kwargs)
                status = 'success'
                return result
            except Exception as e:
                status = 'error'
                raise
            finally:
                response_time = (time.time() - start_time) * 1000
                self.put_metric(
                    'ResponseTime',
                    response_time,
                    'Milliseconds',
                    [{'Name': 'Function', 'Value': func.__name__},
                     {'Name': 'Status', 'Value': status}]
                )
        return wrapper
    
    def track_business_metric(self, metric_name, value, dimensions=None):
        """Rastrear métricas de negócio"""
        self.put_metric(metric_name, value, 'Count', dimensions)
    
    def track_error(self, error_type, error_message):
        """Rastrear erros da aplicação"""
        self.put_metric(
            'ApplicationErrors',
            1,
            'Count',
            [{'Name': 'ErrorType', 'Value': error_type}]
        )
        
        # Log estruturado para CloudWatch Logs
        print(json.dumps({
            'timestamp': datetime.utcnow().isoformat(),
            'level': 'ERROR',
            'error_type': error_type,
            'message': error_message,
            'namespace': self.namespace
        }))

# Instância global
metrics = MetricsCollector()

# app.py com métricas integradas
from flask import Flask, request, jsonify
import pymysql
import json
from metrics import metrics

app = Flask(__name__)

@app.before_request
def before_request():
    """Rastrear requisições recebidas"""
    metrics.put_metric('RequestsReceived', 1, 'Count', [
        {'Name': 'Endpoint', 'Value': request.endpoint or 'unknown'},
        {'Name': 'Method', 'Value': request.method}
    ])

@app.after_request
def after_request(response):
    """Rastrear respostas enviadas"""
    metrics.put_metric('ResponsesSent', 1, 'Count', [
        {'Name': 'StatusCode', 'Value': str(response.status_code)},
        {'Name': 'Endpoint', 'Value': request.endpoint or 'unknown'}
    ])
    return response

@app.route('/api/tasks', methods=['GET'])
@metrics.track_response_time
def get_tasks():
    try:
        # Simular busca de tarefas
        tasks = []  # Implementação da busca
        
        # Métrica de negócio
        metrics.track_business_metric('TasksRetrieved', len(tasks))
        
        return jsonify(tasks)
    except Exception as e:
        metrics.track_error('DatabaseError', str(e))
        return jsonify({'error': 'Internal server error'}), 500

@app.route('/api/health', methods=['GET'])
def health_check():
    """Health check com métricas detalhadas"""
    health_status = {
        'status': 'healthy',
        'timestamp': datetime.utcnow().isoformat(),
        'checks': {}
    }
    
    # Check database
    try:
        # Simular check de banco
        health_status['checks']['database'] = 'healthy'
        metrics.put_metric('HealthCheck', 1, 'Count', [
            {'Name': 'Component', 'Value': 'database'},
            {'Name': 'Status', 'Value': 'healthy'}
        ])
    except:
        health_status['checks']['database'] = 'unhealthy'
        health_status['status'] = 'unhealthy'
        metrics.put_metric('HealthCheck', 1, 'Count', [
            {'Name': 'Component', 'Value': 'database'},
            {'Name': 'Status', 'Value': 'unhealthy'}
        ])
    
    return jsonify(health_status)

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)

## 🚨 Sistema de Alertas Inteligentes

```bash
#!/bin/bash
# alerting/sns-setup.sh

echo "🚨 Configurando sistema de alertas..."

# Criar tópicos SNS
CRITICAL_TOPIC_ARN=$(aws sns create-topic \
    --name "TaskManager-Critical-Alerts" \
    --query 'TopicArn' --output text)

WARNING_TOPIC_ARN=$(aws sns create-topic \
    --name "TaskManager-Warning-Alerts" \
    --query 'TopicArn' --output text)

INFO_TOPIC_ARN=$(aws sns create-topic \
    --name "TaskManager-Info-Alerts" \
    --query 'TopicArn' --output text)

# Configurar subscriptions
aws sns subscribe \
    --topic-arn $CRITICAL_TOPIC_ARN \
    --protocol email \
    --notification-endpoint admin@unifaat.edu.br

aws sns subscribe \
    --topic-arn $CRITICAL_TOPIC_ARN \
    --protocol sms \
    --notification-endpoint +5511999999999

# Webhook para Slack
aws sns subscribe \
    --topic-arn $CRITICAL_TOPIC_ARN \
    --protocol https \
    --notification-endpoint https://hooks.slack.com/services/YOUR/SLACK/WEBHOOK

# Alarmes inteligentes
aws cloudwatch put-metric-alarm \
    --alarm-name "TaskManager-HighErrorRate" \
    --alarm-description "Taxa de erro alta na aplicação" \
    --metric-name "ApplicationErrors" \
    --namespace "UniFAAT/TaskManager" \
    --statistic Sum \
    --period 300 \
    --threshold 10 \
    --comparison-operator GreaterThanThreshold \
    --evaluation-periods 2 \
    --alarm-actions $CRITICAL_TOPIC_ARN \
    --ok-actions $INFO_TOPIC_ARN

# Alarme composto para disponibilidade
aws cloudwatch put-composite-alarm \
    --alarm-name "TaskManager-ServiceUnavailable" \
    --alarm-description "Serviço indisponível - múltiplas métricas" \
    --alarm-rule "(ALARM('TaskManager-HighErrorRate') OR ALARM('TaskManager-HighResponseTime'))" \
    --actions-enabled \
    --alarm-actions $CRITICAL_TOPIC_ARN

echo "✅ Sistema de alertas configurado!"
```

- Tempo de resposta > 5 segundos
- Aplicação indisponível
- Banco de dados inacessível

### Warnings (Email)
- Taxa de erro > 2% em 15 minutos
- Tempo de resposta > 2 segundos
- CPU > 80% por 15 minutos

- Deploy realizado com sucesso
- Backup completado
- Scaling automático ativado
