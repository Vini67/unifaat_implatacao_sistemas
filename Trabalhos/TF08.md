## 🔧 Scripts de Automação

### setup-aws.sh
```bash
#!/bin/bash
set -e

echo "🚀 Configurando ambiente AWS..."

# Verificar AWS CLI
if ! command -v aws &> /dev/null; then
    echo "❌ AWS CLI não encontrado. Instalando..."
    curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
    unzip awscliv2.zip
    sudo ./aws/install
fi

# Configurar profile
echo "⚙️ Configurando AWS CLI..."
aws configure set region us-east-1 --profile default
aws configure set output json --profile default

# Validar configuração
echo "✅ Validando configuração..."
aws sts get-caller-identity

# Criar usuário de desenvolvimento
echo "👤 Criando usuário de desenvolvimento..."
./scripts/create-user.sh developer

# Configurar billing alarms
echo "💰 Configurando alertas de custo..."
./scripts/setup-billing.sh

# Habilitar CloudTrail
echo "📊 Habilitando CloudTrail..."
aws cloudtrail create-trail \
    --name "security-audit-trail" \
    --s3-bucket-name "cloudtrail-logs-$(date +%s)" \
    --include-global-service-events \
    --is-multi-region-trail

echo "✅ Setup AWS concluído!"
```

### validate-config.sh
```bash
#!/bin/bash

echo "🔍 Validando configuração AWS..."

# Verificar identidade
echo "👤 Verificando identidade atual..."
IDENTITY=$(aws sts get-caller-identity)
echo "$IDENTITY"

# Verificar MFA na conta root
echo "🔐 Verificando MFA..."
MFA_DEVICES=$(aws iam list-mfa-devices --user-name root 2>/dev/null || echo "N/A")
if [[ "$MFA_DEVICES" == "N/A" ]]; then
    echo "⚠️ MFA não configurado para usuário atual"
else
    echo "✅ MFA configurado"
fi

# Verificar billing alarms
echo "💰 Verificando billing alarms..."
ALARMS=$(aws cloudwatch describe-alarms --alarm-name-prefix "Billing")
ALARM_COUNT=$(echo "$ALARMS" | jq '.MetricAlarms | length')
echo "📊 Encontrados $ALARM_COUNT billing alarms"

# Verificar CloudTrail
echo "📊 Verificando CloudTrail..."
TRAILS=$(aws cloudtrail describe-trails)
TRAIL_COUNT=$(echo "$TRAILS" | jq '.trailList | length')
echo "🛤️ Encontrados $TRAIL_COUNT trails configurados"

# Verificar usuários IAM
echo "👥 Verificando usuários IAM..."
USERS=$(aws iam list-users)
USER_COUNT=$(echo "$USERS" | jq '.Users | length')
echo "👤 Encontrados $USER_COUNT usuários IAM"

# Relatório final
echo ""
echo "📋 RELATÓRIO DE VALIDAÇÃO"
echo "========================"
echo "✅ AWS CLI: Configurado"
echo "✅ Identidade: Validada"
echo "📊 Billing Alarms: $ALARM_COUNT"
echo "🛤️ CloudTrail: $TRAIL_COUNT"
echo "👤 Usuários IAM: $USER_COUNT"
echo ""

if [[ $ALARM_COUNT -gt 0 && $TRAIL_COUNT -gt 0 ]]; then
    echo "✅ Configuração AWS está adequada!"
    exit 0
else
    echo "⚠️ Configuração AWS precisa de ajustes"
    exit 1
fi
```
