## 🚀 User Data Script

```bash
#!/bin/bash
# User Data Script - Aplicação Web Automatizada
# Versão: 2.0

set -e
exec > >(tee /var/log/user-data.log|logger -t user-data -s 2>/dev/console) 2>&1

echo "=== INICIANDO BOOTSTRAP AUTOMATIZADO ==="
date

# Variáveis
APP_VERSION=${APP_VERSION:-"v1.0.0"}
APP_NAME="webapp-unifaat"
LOG_FILE="/var/log/bootstrap.log"

# Função de log
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a $LOG_FILE
}

# Função de notificação
notify() {
    local status=$1
    local message=$2
    
    # Enviar para CloudWatch Logs
    aws logs put-log-events \
        --log-group-name "/aws/ec2/bootstrap" \
        --log-stream-name "$(curl -s http://169.254.169.254/latest/meta-data/instance-id)" \
        --log-events timestamp=$(date +%s000),message="$status: $message" || true
}

# Atualizar sistema
log "Atualizando sistema..."
dnf update -y

# Instalar dependências
log "Instalando dependências..."
dnf install -y python3 python3-pip nginx git awscli

# Configurar aplicação
log "Configurando aplicação..."
cd /opt
git clone https://github.com/usuario/webapp-unifaat.git
cd webapp-unifaat
git checkout $APP_VERSION

# Instalar dependências Python
pip3 install -r requirements.txt

# Configurar Nginx
log "Configurando Nginx..."
cat > /etc/nginx/conf.d/webapp.conf << 'EOF'
server {
    listen 80;
    server_name _;
    
    location / {
        proxy_pass http://127.0.0.1:5000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
    
    location /health {
        proxy_pass http://127.0.0.1:5000/health;
    }
}
EOF

# Criar serviço systemd
log "Criando serviço systemd..."
cat > /etc/systemd/system/webapp.service << 'EOF'
[Unit]
Description=WebApp UniFAAT
After=network.target

[Service]
Type=simple
User=ec2-user
WorkingDirectory=/opt/webapp-unifaat
ExecStart=/usr/bin/python3 app.py
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target
EOF

# Habilitar e iniciar serviços
log "Iniciando serviços..."
systemctl daemon-reload
systemctl enable webapp nginx
systemctl start webapp nginx

# Aguardar aplicação ficar pronta
log "Aguardando aplicação..."
for i in {1..30}; do
    if curl -f http://localhost/health; then
        log "Aplicação está saudável!"
        notify "SUCCESS" "Bootstrap concluído com sucesso"
        break
    fi
    sleep 10
done

# Configurar monitoramento
log "Configurando monitoramento..."
cat > /opt/health-monitor.sh << 'EOF'
#!/bin/bash
while true; do
    if ! curl -f http://localhost/health > /dev/null 2>&1; then
        echo "ALERT: Aplicação não está respondendo" | logger
        systemctl restart webapp
    fi
    sleep 60
done
EOF

chmod +x /opt/health-monitor.sh
nohup /opt/health-monitor.sh &

log "=== BOOTSTRAP CONCLUÍDO ==="
date
```

## 🏗️ AMI Builder Script

```bash
#!/bin/bash
# build-ami.sh - Construir Golden Image

set -e

VERSION=${1:-$(date +%Y%m%d-%H%M%S)}
AMI_NAME="webapp-unifaat-golden-$VERSION"

echo "🏗️ Construindo AMI: $AMI_NAME"

# Lançar instância temporária
INSTANCE_ID=$(aws ec2 run-instances \
    --image-id ami-0c02fb55956c7d316 \
    --instance-type t2.micro \
    --key-name unifaat-key \
    --security-group-ids sg-12345678 \
    --user-data file://user-data/bootstrap-app.sh \
    --tag-specifications "ResourceType=instance,Tags=[{Key=Name,Value=AMI-Builder-$VERSION}]" \
    --query 'Instances[0].InstanceId' \
    --output text)

echo "📦 Instância criada: $INSTANCE_ID"

# Aguardar instância ficar pronta
echo "⏳ Aguardando instância ficar pronta..."
aws ec2 wait instance-running --instance-ids $INSTANCE_ID

# Aguardar bootstrap completar
echo "⏳ Aguardando bootstrap completar..."
sleep 300

# Validar aplicação
INSTANCE_IP=$(aws ec2 describe-instances \
    --instance-ids $INSTANCE_ID \
    --query 'Reservations[0].Instances[0].PublicIpAddress' \
    --output text)

echo "🔍 Validando aplicação em $INSTANCE_IP..."
for i in {1..10}; do
    if curl -f http://$INSTANCE_IP/health; then
        echo "✅ Aplicação validada!"
        break
    fi
    sleep 30
done

# Parar instância
echo "⏹️ Parando instância..."
aws ec2 stop-instances --instance-ids $INSTANCE_ID
aws ec2 wait instance-stopped --instance-ids $INSTANCE_ID

# Criar AMI
echo "📸 Criando AMI..."
AMI_ID=$(aws ec2 create-image \
    --instance-id $INSTANCE_ID \
    --name "$AMI_NAME" \
    --description "Golden Image - WebApp UniFAAT v$VERSION" \
    --no-reboot \
    --tag-specifications "ResourceType=image,Tags=[{Key=Version,Value=$VERSION},{Key=Application,Value=webapp-unifaat}]" \
    --query 'ImageId' \
    --output text)

echo "🎉 AMI criada: $AMI_ID"

# Aguardar AMI ficar disponível
echo "⏳ Aguardando AMI ficar disponível..."
aws ec2 wait image-available --image-ids $AMI_ID

# Limpar instância temporária
echo "🧹 Limpando instância temporária..."
aws ec2 terminate-instances --instance-ids $INSTANCE_ID

echo "✅ AMI $AMI_ID criada com sucesso!"
echo "📝 Para usar: aws ec2 run-instances --image-id $AMI_ID"
```
